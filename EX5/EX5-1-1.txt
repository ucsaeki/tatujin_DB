EX5-1-1 商品分類ごとの商品数を求める。

/*商品テーブル、商品分類テーブルを内部結合し、商品分類コードごとの商品の個数を数える。*/


SELECT
	c.goods_category_code AS '商品分類コード',
	c.goods_category AS '分類名',
	COUNT(g.goods_category_code) AS '商品数'
FROM
	goods_categories AS c
INNER JOIN
	goods AS g
ON
	c.goods_category_code = g.goods_category_code
GROUP BY
	c.goods_category_code
;
