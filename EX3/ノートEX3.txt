CREATE TABLE
	branch_office_goods
	(
		branch_office_code CHAR(3),
		branch_code CHAR(2),
		goods_code CHAR(3),
		PRIMARY KEY (branch_office_code,branch_code,goods_code)
	)
;


CREATE TABLE
	branch_office_goods
	(
		goods_category_code CHAR(2),
		goods_category VARCHAR(20),
		PRIMARY KEY(goods_category_code)
	)
;



SELECT
	branch_office_code AS '支社コード',
	branch_code AS '支店コード',
	goods_code AS '商品コード'
FROM
	branch_office_goods
;
